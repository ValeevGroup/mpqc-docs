Detailed Description

template<typename Tile, typename Sink = detail::default_sink>
class mpqc::lcao::gaussian::DirectContract< Tile, Sink >

`‘contracts’' a tile of a given DistArray (or a sequence of DistArray's) with a tile of gaussian AO integrals

\/\/ TA::DistArray<T,P> D;
\/\/ contracts D(p,q) with (p|V|q)
DirectContract<T> op1(bases, gaussian::make_engine_pool(libint2::Operator::nuclear, bases, libint2::BraKet::xs_xx));
TA::foreach_inplace(D, op1);
\/\/ contracts D(p,q) with (X|pq)
DirectContract<T> op2(bases, gaussian::make_engine_pool(libint2::Operator::coulomb, bases, libint2::BraKet::xs_xx), {0,1}, {2,0,1}, {2});
TA::foreach_inplace(D, op2);
