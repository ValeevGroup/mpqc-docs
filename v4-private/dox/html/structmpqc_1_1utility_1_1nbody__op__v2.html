template<std::size_t K, typename T, ParticleStatistics Statistics = ParticleStatistics::FermiDirac>
struct mpqc::utility::nbody_op_v2< K, T, Statistics >

n-body operator composed of up to K-body ops using ManyBodyOperatorTensors<K> rather than individual tensors
